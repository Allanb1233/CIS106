#define the object
class Employee:
  # use pass to leave empty for now
  def init(self, first, last, pay):
    self.first = first
    self.last = last
    self.pay = pay
    #self.rate = 0.00

  def bonus(self,bonusrate):
    b = float(bonusrate) * float(self.pay)
    return b

class Manager(Employee):
  def mbonus(self,mgrbonus):
    b = float(mgrbonus) * float(self.pay)
    return b
#instantiate the object

empl1 = Employee('John', 'Smith',50000)
mana1 = Manager('John', 'Smith',50000)
bonusrate = input("Enter bonus rate: ")
mgrbonus = 0.4
print(empl1.first)
print(empl1.last)
print(empl1.bonus(bonusrate))
print(mana1.mbonus(mgrbonus))
